Aquí encontrarás información sobre derechos, salud y bienestar."; if (speechSynthesis.speaking) { speechSynthesis.cancel(); } const utter = new SpeechSynthesisUtterance(text); utter.lang = 'es-MX'; utter.rate = 0.95; utter.onstart = () => { dog.play(); dog.classList.add('dog-speaking'); }; utter.onend = () => { dog.pause(); dog.classList.remove('dog-speaking'); dog.currentTime = 0; }; speechSynthesis.speak(utter); }
